✦ LumiSearch Plugin v2

AI-powered search
for any website

Drop one script tag. Get a beautiful search popup with product cards, intent routing, and a persistent chat window — no framework required.

Angular Integration

app.component.ts 
// angular.json → scripts:
"src/assets/lumisearch-plugin.js"

// app.component.ts
import { LumiSearchService } from './services/lumisearch.service';

export class AppComponent implements AfterViewInit {
  constructor(private lumi: LumiSearchService) {}

  ngAfterViewInit() {
    this.lumi.init({
      accentColor: '#1a56db',
      brandName:   'Luminous AI',
    });
  }
}

// Any child component
openSearch() { this.lumi.open(); }
